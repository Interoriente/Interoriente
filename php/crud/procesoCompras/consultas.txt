SELECT PU.nombrePublicacion AS "titulo", CA.cantidadCarrito as "cantidad", PU.costoPublicacion AS "costo", SUM(CA.cantidadCarrito * PU.costoPublicacion) AS "subtotal", (SUM(CA.cantidadCarrito * PU.costoPublicacion) * 0.19) AS 'iva', (SUM(CA.cantidadCarrito * PU.costoPublicacion) + (SUM(CA.cantidadCarrito * PU.costoPublicacion) * 0.19)) AS "total"
FROM tblPublicacion AS PU INNER JOIN tblCarrito AS CA ON PU.idPublicacion = CA.idPublicacionCarrito
INNER JOIN tblUsuario AS US ON CA.docIdentidadCarrito = US.documentoIdentidad
GROUP BY PU.idPublicacion
